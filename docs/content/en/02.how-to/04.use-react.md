## Install

```bash
yarn add @json-eval-rs/bundler
```

Use public bundler wrapper. Do not import internal `@json-eval-rs/webcore` or pass a `wasmModule` constructor option.

## Create evaluator hook

```tsx
import { useEffect, useState } from 'react';
import type { JSONEval as JSONEvalType } from '@json-eval-rs/bundler';

export function useJSONEval(schema: object) {
  const [evaluator, setEvaluator] = useState<JSONEvalType | null>(null);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    let instance: JSONEvalType | null = null;
    let active = true;

    async function load() {
      try {
        const { JSONEval } = await import('@json-eval-rs/bundler');
        instance = new JSONEval({ schema });
        await instance.init();
        if (active) setEvaluator(instance);
      } catch (cause) {
        if (active) setError(cause instanceof Error ? cause : new Error(String(cause)));
      }
    }

    load();
    return () => {
      active = false;
      instance?.free();
    };
  }, [schema]);

  return { evaluator, error };
}
```

Keep `schema` reference stable (`useMemo` or module constant) to avoid rebuilding evaluator every render.

## Validate form input

```tsx
const { evaluator, error } = useJSONEval(schema);
const [validation, setValidation] = useState<any>(null);

async function validate(age: string) {
  if (!evaluator) return;
  setValidation(await evaluator.validate({
    data: { age: Number(age) || 0 }
  }));
}
```

Use `evaluate({ data, paths })` for computed schema; use `validatePaths({ data, paths })` when validating selected fields only.

## Cleanup

`JSONEval` owns WASM resources. Call `free()` once for each instance, normally in `useEffect` cleanup. Do not free while evaluation still uses instance.

For package choice and compact/resolved output, see [Web & NodeJS](use-web-node).
