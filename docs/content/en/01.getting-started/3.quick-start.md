Use a schema containing one validation rule. Each example evaluates current data, then validates it.

## Rust

```rust
use json_eval_rs::JSONEval;

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let schema = r#"{
      "type": "object",
      "properties": {
        "age": {
          "rules": {
            "minValue": { "value": 18, "message": "Must be 18 or older" }
          }
        }
      }
    }"#;
    let data = r#"{"age":25}"#;

    let mut evaluator = JSONEval::new(schema, None, None)?;
    evaluator.evaluate(data, None, None, None)?;

    let validation = evaluator.validate(data, None, None)?;
    assert!(!validation.has_error);
    Ok(())
}
```

## C# / .NET

```csharp
using JsonEvalRs;

var schema = @"{
  ""type"": ""object"",
  ""properties"": {
    ""age"": {
      ""rules"": {
        ""minValue"": { ""value"": 18, ""message"": ""Must be 18 or older"" }
      }
    }
  }
}";
var data = @"{""age"":25}";

using var evaluator = new JSONEval(schema);
evaluator.Evaluate(data);
var validation = evaluator.Validate(data);
Console.WriteLine(validation.HasError ? "Invalid" : "Valid");
```

## Node.js, Vite, Webpack, or Next.js

Import `JSONEval` from runtime package directly. Methods initialize WebAssembly on first async call.

```typescript
import { JSONEval } from '@json-eval-rs/node';
// Use '@json-eval-rs/bundler' instead in Vite, Webpack, Rollup, or Next.js.

const schema = {
  type: 'object',
  properties: {
    age: {
      rules: {
        minValue: { value: 18, message: 'Must be 18 or older' },
      },
    },
  },
};

const evaluator = new JSONEval({ schema });
try {
  await evaluator.evaluate({ data: { age: 25 } });
  const validation = await evaluator.validate({ data: { age: 25 } });
  console.log(validation.has_error ? 'Invalid' : 'Valid');
} finally {
  evaluator.free();
}
```

## Browser without bundler

Initialize `@json-eval-rs/vanilla` once before making calls.

```typescript
import { JSONEval } from '@json-eval-rs/vanilla';

await JSONEval.initWasm();
const evaluator = new JSONEval({ schema });
try {
  const validation = await evaluator.validate({ data: { age: 25 } });
  console.log(validation.has_error ? 'Invalid' : 'Valid');
} finally {
  evaluator.free();
}
```

## React Native

`useJSONEval` creates evaluator after mount and disposes it on unmount.

```tsx
import { useState } from 'react';
import { Text, TextInput, View } from 'react-native';
import { useJSONEval } from '@json-eval-rs/react-native';

const schema = {
  type: 'object',
  properties: {
    age: { rules: { minValue: { value: 18, message: 'Must be 18 or older' } } },
  },
};

export function AgeValidator() {
  const evaluator = useJSONEval({ schema });
  const [message, setMessage] = useState('');

  async function validate(age: string) {
    if (!evaluator) return;
    const result = await evaluator.validate({ data: { age: Number(age) || 0 } });
    setMessage(result.has_error ? 'Invalid' : 'Valid');
  }

  return <View><TextInput keyboardType="numeric" onChangeText={validate} /><Text>{message}</Text></View>;
}
```
