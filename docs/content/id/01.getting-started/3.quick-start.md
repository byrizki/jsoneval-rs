Gunakan schema dengan satu rule validasi. Setiap contoh mengevaluasi data saat ini, lalu memvalidasinya.

## Rust

```rust
use json_eval_rs::JSONEval;

fn main() -> Result<(), Box<dyn std::error::Error>> {
    let schema = r#"{
      "type": "object",
      "properties": {
        "age": {
          "rules": {
            "minValue": { "value": 18, "message": "Harus berumur 18 atau lebih" }
          }
        }
      }
    }"#;
    let data = r#"{"age":25}"#;

    let mut evaluator = JSONEval::new(schema, None, None)?;
    evaluator.evaluate(data, None, None, None)?;

    let validation = evaluator.validate(data, None, None)?;
    assert!(!validation.has_error);
    Ok(())
}
```

## C# / .NET

```csharp
using JsonEvalRs;

var schema = @"{
  ""type"": ""object"",
  ""properties"": {
    ""age"": {
      ""rules"": {
        ""minValue"": { ""value"": 18, ""message"": ""Harus berumur 18 atau lebih"" }
      }
    }
  }
}";
var data = @"{""age"":25}";

using var evaluator = new JSONEval(schema);
evaluator.Evaluate(data);
var validation = evaluator.Validate(data);
Console.WriteLine(validation.HasError ? "Tidak valid" : "Valid");
```

## Node.js, Vite, Webpack, atau Next.js

Import `JSONEval` langsung dari paket runtime. Method async menginisialisasi WebAssembly saat pertama dipanggil.

```typescript
import { JSONEval } from '@json-eval-rs/node';
// Pakai '@json-eval-rs/bundler' di Vite, Webpack, Rollup, atau Next.js.

const schema = {
  type: 'object',
  properties: {
    age: {
      rules: {
        minValue: { value: 18, message: 'Harus berumur 18 atau lebih' },
      },
    },
  },
};

const evaluator = new JSONEval({ schema });
try {
  await evaluator.evaluate({ data: { age: 25 } });
  const validation = await evaluator.validate({ data: { age: 25 } });
  console.log(validation.has_error ? 'Tidak valid' : 'Valid');
} finally {
  evaluator.free();
}
```

## Browser tanpa bundler

Inisialisasi `@json-eval-rs/vanilla` sekali sebelum memanggil method lain.

```typescript
import { JSONEval } from '@json-eval-rs/vanilla';

await JSONEval.initWasm();
const evaluator = new JSONEval({ schema });
try {
  const validation = await evaluator.validate({ data: { age: 25 } });
  console.log(validation.has_error ? 'Tidak valid' : 'Valid');
} finally {
  evaluator.free();
}
```

## React Native

`useJSONEval` membuat evaluator setelah mount dan membuangnya saat unmount.

```tsx
import { useState } from 'react';
import { Text, TextInput, View } from 'react-native';
import { useJSONEval } from '@json-eval-rs/react-native';

const schema = {
  type: 'object',
  properties: {
    age: { rules: { minValue: { value: 18, message: 'Harus berumur 18 atau lebih' } } },
  },
};

export function AgeValidator() {
  const evaluator = useJSONEval({ schema });
  const [message, setMessage] = useState('');

  async function validate(age: string) {
    if (!evaluator) return;
    const result = await evaluator.validate({ data: { age: Number(age) || 0 } });
    setMessage(result.has_error ? 'Tidak valid' : 'Valid');
  }

  return <View><TextInput keyboardType="numeric" onChangeText={validate} /><Text>{message}</Text></View>;
}
```
